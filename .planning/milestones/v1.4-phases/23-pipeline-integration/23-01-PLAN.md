```groovy
#!/usr/bin/env groovy
// Noda Apps 蓝绿部署 Pipeline
// 8 阶段: Pre-flight -> Build -> Test -> Deploy -> Health Check -> Switch -> Verify -> Cleanup
// 手动触发（无自动触发配置）
// Phase 23: Pipeline 集成与测试门禁

pipeline {
    agent any

    environment {
        PROJECT_ROOT = "${WORKSPACE}"
        ACTIVE_ENV = sh(
            script: 'cat /opt/noda/active-env 2>/dev/null || echo blue',
            returnStdout: true
        ).trim()
        TARGET_ENV = "${env.ACTIVE_ENV == 'blue' ? 'green' : 'blue'}"
    }

    options {
        // 禁止并发构建，防止两个 Pipeline 同时操作蓝绿容器
        disableConcurrentBuilds()
        // 保留最近 20 次构建日志
        buildDiscarder(logRotator(numToKeepStr: '20'))
        timestamps()
    }

    // 注意：无 triggers 块 — 仅通过 "Build Now" 手动触发 (per D-04/PIPE-04)

    stages {
        stage('Pre-flight') {
            steps {
                // checkout noda-apps 仓库到子目录
                dir('noda-apps') {
                    checkout([
                        $class: 'GitSCM',
                        branches: [[name: '*/main']],
                        userRemoteConfigs: [[
                            url: 'git@github.com:dianwenwang/noda-apps.git',
                            credentialsId: 'noda-apps-git-credentials'
                        ]]
                    ])
                }
                // 获取 noda-apps 的 Git SHA（workspace 根是 noda-infra）
                script {
                    env.GIT_SHA = sh(
                        script: 'git -C noda-apps rev-parse --short HEAD',
                        returnStdout: true
                    ).trim()
                }
                // 前置检查 — 单一真相源：所有检查逻辑在 pipeline_preflight 中
                // 不在此处写内联 Docker/nginx/network/pnpm 检查
                sh '''
                    source scripts/lib/log.sh
                    source scripts/pipeline-stages.sh
                    pipeline_preflight
                '''
            }
        }

        stage('Build') {
            steps {
                sh '''
                    source scripts/lib/log.sh
                    source scripts/pipeline-stages.sh
                    pipeline_build "$WORKSPACE/noda-apps" "$GIT_SHA"
                '''
            }
        }

        stage('Test') {
            steps {
                dir('noda-apps') {
                    // 依赖安装由 pipeline_test 处理
                    sh '''
                        source "$WORKSPACE/scripts/lib/log.sh"
                        source "$WORKSPACE/scripts/pipeline-stages.sh"
                        pipeline_test "$PWD"
                    '''
                    // lint 和 test 为独立 sh 步骤（RESEARCH.md Pitfall 4）
                    // 这样 Jenkins Stage View 可区分是 lint 还是 test 失败
                    // 不通过 pipeline_test 函数调用，直接 sh 步骤确保 Stage View 粒度
                    sh 'pnpm lint'
                    sh 'pnpm test'
                }
            }
        }

        stage('Deploy') {
            steps {
                sh '''
                    source scripts/lib/log.sh
                    source scripts/pipeline-stages.sh
                    pipeline_deploy "$TARGET_ENV" "$GIT_SHA"
                '''
            }
        }

        stage('Health Check') {
            steps {
                sh '''
                    source scripts/lib/log.sh
                    source scripts/pipeline-stages.sh
                    pipeline_health_check "$TARGET_ENV"
                '''
            }
        }

        stage('Switch') {
            steps {
                sh '''
                    source scripts/lib/log.sh
                    source scripts/pipeline-stages.sh
                    pipeline_switch "$TARGET_ENV" "$ACTIVE_ENV"
                '''
            }
        }

        stage('Verify') {
            steps {
                sh '''
                    source scripts/lib/log.sh
                    source scripts/pipeline-stages.sh
                    pipeline_verify "$TARGET_ENV"
                '''
            }
        }

        stage('Cleanup') {
            steps {
                sh '''
                    source scripts/lib/log.sh
                    source scripts/pipeline-stages.sh
                    pipeline_cleanup
                '''
            }
        }
    }

    post {
        failure {
            script {
                // 检查 TARGET_ENV 是否已定义（Pitfall 5: 可能 Pre-flight 就失败了）
                if (env.TARGET_ENV && env.TARGET_ENV != 'null') {
                    sh '''
                        source scripts/lib/log.sh
                        source scripts/pipeline-stages.sh
                        pipeline_failure_cleanup "$TARGET_ENV"
                    '''
                    archiveArtifacts artifacts: 'deploy-failure-*.log', allowEmptyArchive: true
                } else {
                    echo 'Pipeline 在 TARGET_ENV 设置前失败，跳过日志归档'
                }
            }
        }
        success {
            echo "部署成功: ${env.TARGET_ENV} 现在是活跃环境 (镜像: findclass-ssr:${env.GIT_SHA})"
        }
    }
}
```

关键设计说明：
- **单引号 sh 命令**: 所有 `sh '''...'''` 使用单引号，防止 Groovy 插值（RESEARCH.md Anti-Pattern）。只有需要 Jenkins 环境变量插值的地方用双引号 `"..."`。
- **pnpm install/lint/test 三步分离**: Test 阶段中 install（通过 pipeline_test）、lint、test 是三个独立 sh 步骤（RESEARCH.md Pitfall 4），Stage View 可区分是哪一步失败。
- **Pre-flight 单一真相源**: Pre-flight stage 不包含任何内联的 Docker/nginx/network 检查，所有检查逻辑集中在 `pipeline_preflight()` 函数中，避免两处维护。
- **post failure 防护**: 检查 `env.TARGET_ENV` 是否已定义，防止 Pre-flight 失败时 `docker logs findclass-ssr-null`（RESEARCH.md Pitfall 5）。
