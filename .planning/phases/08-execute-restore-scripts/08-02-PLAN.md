test_list_backups() {
  test_start 1 "列出 B2 上所有可用的备份文件，按时间排序"

  local output
  output=$(bash "$SCRIPT_DIR/restore-postgres.sh" --list-backups 2>&1) || true

  # 验证输出包含备份文件信息
  if echo "$output" | grep -qE '[0-9]{4}-[0-9]{2}-[0-9]{2}'; then
    # 验证包含数据库名
    if echo "$output" | grep -qE '\.(dump|sql)'; then
      test_pass "列出备份文件"
    else
      test_fail "列出备份文件" "输出中未找到备份文件（.dump/.sql 扩展名）"
    fi
  else
    test_fail "列出备份文件" "输出中未找到日期格式或 B2 连接失败，输出: $(echo "$output" | tail -3)"
  fi
}
```

test_restore_to_different_db() {
  test_start 2 "指定备份文件恢复到目标数据库，支持恢复到不同数据库名"

  # 从 B2 获取最新的 .dump 文件
  local backup_filename
  backup_filename=$(bash "$SCRIPT_DIR/restore-postgres.sh" --list-backups 2>&1 \
    | grep -oE '[a-z_]+_[0-9]{8}_[0-9]{6}\.dump' \
    | head -1) || true

  if [[ -z "$backup_filename" ]]; then
    test_fail "恢复到不同数据库" "未找到可用的 .dump 备份文件"
    return
  fi

  # 使用临时数据库名恢复（D-07, D-13）
  local test_db_name="test_verify_restore_$(date +%s)"

  # 通过管道输入 yes 确认恢复（D-04 确认机制）
  echo "yes" | bash "$SCRIPT_DIR/restore-postgres.sh" --restore "$backup_filename" --database "$test_db_name" 2>&1 || true

  # 验证目标数据库已创建
  local db_exists
  db_exists=$(docker exec noda-infra-postgres-1 psql -U postgres -d postgres -t -c \
    "SELECT 1 FROM pg_database WHERE datname='$test_db_name';" 2>/dev/null | xargs || echo "")

  if [[ "$db_exists" == "1" ]]; then
    # 验证表数量 > 0（D-14，小型数据库验证）
    local table_count
    table_count=$(docker exec noda-infra-postgres-1 psql -U postgres -d "$test_db_name" -t -c \
      "SELECT COUNT(*) FROM information_schema.tables WHERE table_schema='public';" 2>/dev/null | xargs || echo "0")

    if [[ "$table_count" -gt 0 ]]; then
      test_pass "恢复到不同数据库（$table_count 个表）"
    else
      test_fail "恢复到不同数据库" "数据库已创建但表数量为 0"
    fi

    # 清理临时数据库（D-15）
    docker exec noda-infra-postgres-1 psql -U postgres -d postgres -c \
      "DROP DATABASE IF EXISTS $test_db_name;" >/dev/null 2>&1 || true
  else
    test_fail "恢复到不同数据库" "目标数据库未创建"
  fi
}

test_verify_backup_integrity() {
  test_start 3 "恢复前自动验证备份文件完整性"

  # 创建小型测试数据库进行本地验证
  local test_db="test_verify_integrity_$(date +%s)"
  docker exec noda-infra-postgres-1 psql -U postgres -d postgres -c \
    "CREATE DATABASE $test_db;" >/dev/null 2>&1
  docker exec noda-infra-postgres-1 psql -U postgres -d "$test_db" >/dev/null 2>&1 <<SQL
CREATE TABLE test_data (id SERIAL PRIMARY KEY, value TEXT);
INSERT INTO test_data (value) VALUES ('test1'), ('test2');
SQL

  # 创建本地备份用于验证
  local tmp_dir
  tmp_dir=$(mktemp -d)
  docker exec noda-infra-postgres-1 pg_dump -U postgres -Fc "$test_db" > "$tmp_dir/test_backup.dump" 2>/dev/null

  # 使用 verify_backup_integrity 验证
  source "$SCRIPT_DIR/lib/restore.sh"
  if verify_backup_integrity "$tmp_dir/test_backup.dump" 2>&1; then
    test_pass "备份完整性验证（文件大小 + pg_restore --list）"
  else
    test_fail "备份完整性验证" "verify_backup_integrity 返回失败"
  fi

  # 清理（D-15）
  docker exec noda-infra-postgres-1 psql -U postgres -d postgres -c \
    "DROP DATABASE IF EXISTS $test_db;" >/dev/null 2>&1 || true
  rm -rf "$tmp_dir"
}
```

## 4. 使用指南

### 基本用法

列出所有可用备份:
```bash
./scripts/backup/restore-postgres.sh --list-backups
```

恢复指定备份到原数据库:
```bash
./scripts/backup/restore-postgres.sh --restore keycloak_db_20260406_081638.dump
```

恢复到不同数据库名（用于安全测试）:
```bash
./scripts/backup/restore-postgres.sh --restore keycloak_db_20260406_081638.dump --database keycloak_db_test
```

仅验证备份完整性（不执行恢复）:
```bash
./scripts/backup/restore-postgres.sh --restore keycloak_db_20260406_081638.dump --verify
```

下载备份到指定目录:
```bash
./scripts/backup/restore-postgres.sh --restore keycloak_db_20260406_081638.dump --output-dir /tmp/backups
```

### 最佳实践
- 恢复前始终先使用 --list-backups 确认可用备份
- 恢复到生产数据库前，先使用 --database 参数恢复到测试库验证数据完整性
- 使用 --verify 参数单独验证备份文件，无需执行完整恢复
- 恢复操作会覆盖目标数据库，请谨慎选择目标数据库名

- Docker 运行中且 noda-infra-postgres-1 容器健康
- rclone 已安装（B2 云操作）
- .env.backup 中 B2 凭证配置正确
