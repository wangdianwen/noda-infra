   local deploy_image=""
   case "$IMAGE_SOURCE" in
     build)
       local short_sha
       short_sha=$(git -C "$apps_dir" rev-parse --short HEAD 2>/dev/null || true)
       if [ -z "$short_sha" ]; then
         log_error "无法获取 Git SHA（目录: $apps_dir），请确认在 git 仓库中执行"
