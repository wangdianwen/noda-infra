具体修改：

**1. 新增 pipeline_pull_image 函数（在 pipeline_build 函数之后插入）**

```bash
# ============================================
# 函数: pipeline_pull_image
# ============================================
# 拉取官方镜像（用于不从源码构建的服务如 Keycloak）
# 环境变量控制：
#   SERVICE_IMAGE - 官方镜像名（如 quay.io/keycloak/keycloak:26.2.3）
# 返回：0=成功，1=失败
pipeline_pull_image() {
  local image="${SERVICE_IMAGE:-}"

  if [ -z "$image" ]; then
    log_error "SERVICE_IMAGE 未设置，无法拉取镜像"
    return 1
  fi

  log_info "拉取镜像: $image"

  if ! docker pull "$image"; then
    log_error "镜像拉取失败: $image"
    return 1
  fi

  log_success "镜像拉取完成: $image"
}
```

在 `pipeline_preflight` 函数中，`if [ "$service" = "findclass-ssr" ]` 分支之后，`else` 分支中添加 Keycloak 专用逻辑：

当前 else 分支只检查 Dockerfile：
```bash

    local dockerfile="${DOCKERFILE:-$PROJECT_ROOT/deploy/Dockerfile.${service}}"

```

改为：

**3. 修改 pipeline_deploy 函数 — 支持使用官方镜像名（非 Git SHA 标签）**

当前 pipeline_deploy 使用 `${service}:${git_sha}` 作为镜像名。Keycloak 不需要 Git SHA，直接使用 SERVICE_IMAGE。

修改 pipeline_deploy：
```bash
pipeline_deploy() {
  local target_env="$1"
  local git_sha="${2:-}"
  local service="${SERVICE_NAME:-findclass-ssr}"
  local target_container
  target_container=$(get_container_name "$target_env")

  # 确定镜像名：如果设置了 SERVICE_IMAGE 则使用官方镜像，否则使用 Git SHA 标签
  local image_name
  if [ -n "${SERVICE_IMAGE:-}" ]; then
    image_name="$SERVICE_IMAGE"
  else
    image_name="${service}:${git_sha}"
  fi

  # 停止旧的无后缀容器（从单容器模式迁移到蓝绿模式）
  local legacy_container="$service"
  if [ "$(is_container_running "$legacy_container")" = "true" ] \
     && [ "$legacy_container" != "$target_container" ]; then
    log_info "停止旧容器: $legacy_container"
    docker stop -t 30 "$legacy_container"
    docker rm "$legacy_container"
  fi

  # 停止旧目标容器（同色蓝绿容器）
  if [ "$(is_container_running "$target_container")" = "true" ]; then
    log_info "停止旧目标容器: $target_container"
    docker stop -t 30 "$target_container"
    docker rm "$target_container"
  fi

  # 启动新容器
  run_container "$target_env" "$image_name"
  log_success "部署完成: $target_container ($image_name)"
}
```

**4. 修改 pipeline_cleanup 函数 — Keycloak 不需要 SHA 镜像清理**

当前 pipeline_cleanup 调用 `cleanup_old_images`，这个函数按 Git SHA 标签清理。Keycloak 使用固定版本标签，不需要此清理。

修改方案：在 cleanup_old_images 之前检查 SERVICE_IMAGE，如果存在则只做 dangling images 清理：

```bash
pipeline_cleanup() {
  # 停掉非活跃容器，降低资源消耗
  local active_env
  active_env=$(get_active_env)
  local inactive_env
  if [ "$active_env" = "blue" ]; then
    inactive_env="green"
  else
    inactive_env="blue"
  fi
  local inactive_container
  inactive_container=$(get_container_name "$inactive_env")

  if [ "$(is_container_running "$inactive_container")" = "true" ]; then
    log_info "停止非活跃容器: $inactive_container"
    docker stop -t 10 "$inactive_container"
    docker rm "$inactive_container"
    log_success "非活跃容器已清理: $inactive_container"
  else
    log_info "无非活跃容器需要清理"
  fi

  # 官方镜像服务（Keycloak 等）不需要 SHA 镜像清理
  if [ -z "${SERVICE_IMAGE:-}" ]; then
    cleanup_old_images
  else
    # 仅清理 dangling images
    local dangling_ids
    dangling_ids=$(docker images -f "dangling=true" --format '{{.ID}}' 2>/dev/null || true)
    local count=0
    for img_id in $dangling_ids; do
      docker rmi "$img_id" 2>/dev/null || true
      count=$((count + 1))
    done
    if [ "$count" -gt 0 ]; then
      log_success "dangling 镜像清理完成: 删除 ${count} 个"
    else
      log_info "镜像清理: 无需清理"
    fi
  fi
}
```

7 阶段 Pipeline：Pre-flight -> Pull Image -> Deploy -> Health Check -> Switch -> Verify -> CDN Purge
无 Build, 无 Test, 无 Cleanup 阶段（Keycloak 使用官方镜像，不需要构建/测试/镜像清理）。

```groovy
#!/usr/bin/env groovy
// Keycloak 蓝绿部署 Pipeline
// 7 阶段: Pre-flight -> Pull Image -> Deploy -> Health Check -> Switch -> Verify -> CDN Purge
// 手动触发（无自动触发配置）
// Keycloak 使用官方镜像，不需要 Build/Test 阶段

pipeline {
    agent any

    environment {
        PROJECT_ROOT = "${WORKSPACE}"
        // 服务参数（覆盖 manage-containers.sh 和 pipeline-stages.sh 的默认值）
        SERVICE_NAME = "keycloak"
        SERVICE_PORT = "8080"
        UPSTREAM_NAME = "keycloak_backend"
        HEALTH_PATH = "/health/ready"
        ACTIVE_ENV_FILE = "/opt/noda/active-env-keycloak"
        UPSTREAM_CONF = "${WORKSPACE}/config/nginx/snippets/upstream-keycloak.conf"
        SERVICE_GROUP = "infra"
        // Keycloak 官方镜像（不从源码构建）
        SERVICE_IMAGE = "quay.io/keycloak/keycloak:26.2.3"
        // Keycloak 容器参数（覆盖 run_container 默认值）
        CONTAINER_MEMORY = "1g"
        CONTAINER_MEMORY_RESERVATION = "512m"
        CONTAINER_READONLY = "false"
        // envsubst 变量列表
        ENVSUBST_VARS = '${POSTGRES_USER} ${POSTGRES_PASSWORD} ${KEYCLOAK_ADMIN_USER} ${KEYCLOAK_ADMIN_PASSWORD} ${SMTP_HOST} ${SMTP_PORT} ${SMTP_FROM} ${SMTP_USER} ${SMTP_PASSWORD}'
        // 额外 docker run 参数（主题卷 + data tmpfs）
        EXTRA_DOCKER_ARGS = "-v ${WORKSPACE}/docker/services/keycloak/themes:/opt/keycloak/themes/noda:ro --tmpfs /opt/keycloak/data"
        // 蓝绿状态
        ACTIVE_ENV = sh(
            script: 'cat /opt/noda/active-env-keycloak 2>/dev/null || echo blue',
            returnStdout: true
        ).trim()
        TARGET_ENV = "${env.ACTIVE_ENV == 'blue' ? 'green' : 'blue'}"
    }

    options {
        // 禁止并发构建，防止两个 Pipeline 同时操作蓝绿容器
        disableConcurrentBuilds()
        // 保留最近 20 次构建日志
        buildDiscarder(logRotator(numToKeepStr: '20'))
        timestamps()
    }

    // 注意：无 triggers 块 — 仅通过 "Build Now" 手动触发

    stages {
        stage('Pre-flight') {
            steps {
                // Keycloak 不需要 checkout noda-apps（使用官方镜像）
                sh '''
                    source scripts/lib/log.sh
                    source scripts/pipeline-stages.sh
                    pipeline_preflight
                '''
            }
        }

        // 无 Build 阶段 — Keycloak 使用官方镜像 quay.io/keycloak/keycloak:26.2.3
        // 无 Test 阶段 — Keycloak 不从源码构建

        stage('Pull Image') {
            steps {
                sh '''
                    source scripts/lib/log.sh
                    source scripts/pipeline-stages.sh
                    pipeline_pull_image
                '''
            }
        }

        stage('Deploy') {
            steps {
                sh '''
                    source scripts/lib/log.sh
                    source scripts/pipeline-stages.sh
                    pipeline_deploy "$TARGET_ENV"
                '''
            }
        }

        stage('Health Check') {
            steps {
                sh '''
                    source scripts/lib/log.sh
                    source scripts/pipeline-stages.sh
                    pipeline_health_check "$TARGET_ENV"
                '''
            }
        }

        stage('Switch') {
            steps {
                sh '''
                    source scripts/lib/log.sh
                    source scripts/pipeline-stages.sh
                    pipeline_switch "$TARGET_ENV" "$ACTIVE_ENV"
                '''
            }
        }

        stage('Verify') {
            steps {
                sh '''
                    source scripts/lib/log.sh
                    source scripts/pipeline-stages.sh
                    pipeline_verify "$TARGET_ENV"
                '''
            }
        }

        stage('CDN Purge') {
            steps {
                script {
                    // 凭据不存在时优雅跳过
                    try {
                        withCredentials([
                            string(credentialsId: 'cf-api-token', variable: 'CF_API_TOKEN'),
                            string(credentialsId: 'cf-zone-id', variable: 'CF_ZONE_ID')
                        ]) {
                            sh '''
                                source scripts/lib/log.sh
                                source scripts/pipeline-stages.sh
                                pipeline_purge_cdn
                            '''
                        }
                    } catch (Exception e) {
                        echo "CDN Purge 跳过: Cloudflare 凭据未配置 (${e.message})"
                    }
                }
            }
        }

        // 无 Cleanup 阶段 — Keycloak 使用固定版本镜像，不需要 SHA 镜像清理
        // 非活跃容器在下次部署时自动清理（pipeline_deploy 会停止旧目标容器）
    }

    post {
        failure {
            script {
                if (env.TARGET_ENV && env.TARGET_ENV != 'null') {
                    sh '''
                        source scripts/lib/log.sh
                        source scripts/pipeline-stages.sh
                        pipeline_failure_cleanup "$TARGET_ENV"
                    '''
                    archiveArtifacts artifacts: 'deploy-failure-*.log', allowEmptyArchive: true
                } else {
                    echo 'Pipeline 在 TARGET_ENV 设置前失败，跳过日志归档'
                }
            }
        }
        success {
            echo "Keycloak 部署成功: ${env.TARGET_ENV} 现在是活跃环境 (镜像: ${env.SERVICE_IMAGE})"
        }
    }
}
```
