**步骤 2: 触发 noda-ops Pipeline**
```bash
source scripts/jenkins/config/jenkins-admin.env
JENKINS_URL="http://localhost:8888"

# 获取 CSRF Crumb
CRUMB=$(curl -sf -u "$JENKINS_ADMIN_USER:$JENKINS_ADMIN_PASSWORD" \
  "$JENKINS_URL/crumbIssuer/api/json" | \
  python3 -c "import sys,json; d=json.load(sys.stdin); print(d['crumbRequestField']+':'+d['crumb'])")

# 获取当前构建号
NEXT_BUILD=$(curl -sf -u "$JENKINS_ADMIN_USER:$JENKINS_ADMIN_PASSWORD" \
  "$JENKINS_URL/job/infra-deploy/api/json" | \
  python3 -c "import sys,json; d=json.load(sys.stdin); print(d['nextBuildNumber'])")

# 触发 noda-ops 部署
curl -sf -u "$JENKINS_ADMIN_USER:$JENKINS_ADMIN_PASSWORD" \
  -X POST -H "$CRUMB" \
  "$JENKINS_URL/job/infra-deploy/build-withParameters?SERVICE=noda-ops"

BUILD_NUM=$NEXT_BUILD
echo "noda-ops 构建已排队: #$BUILD_NUM"
```

**步骤 3: 等待 noda-ops Pipeline 完成并检查日志**
```bash
# 轮询构建状态（每 30 秒检查一次，最多 10 分钟）
for i in $(seq 1 20); do
  RESULT=$(curl -sf -u "$JENKINS_ADMIN_USER:$JENKINS_ADMIN_PASSWORD" \
    "$JENKINS_URL/job/infra-deploy/$BUILD_NUM/api/json" | \
    python3 -c "import sys,json; d=json.load(sys.stdin); print('building:', d['building'], 'result:', d.get('result','running'))")
  echo "[$i] $RESULT"
  if echo "$RESULT" | grep -q "building: False"; then
    break
  fi
  sleep 30
done

# 获取构建日志中的清理相关输出
curl -sf -u "$JENKINS_ADMIN_USER:$JENKINS_ADMIN_PASSWORD" \
  "$JENKINS_URL/job/infra-deploy/$BUILD_NUM/consoleText" | \
  grep -A5 -E "镜像清理|cleanup_by_date|noda-ops|dangling|清理后|部署后清理"

# 检查构建是否成功
BUILD_RESULT=$(curl -sf -u "$JENKINS_ADMIN_USER:$JENKINS_ADMIN_PASSWORD" \
  "$JENKINS_URL/job/infra-deploy/$BUILD_NUM/api/json" | \
  python3 -c "import sys,json; d=json.load(sys.stdin); print(d.get('result','UNKNOWN'))")
echo "noda-ops 构建结果: $BUILD_RESULT"
```
