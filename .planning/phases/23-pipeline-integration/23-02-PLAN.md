**1. 增强 pipeline_preflight() — 添加 Node.js/pnpm/noda-apps 检查**

在现有的 Docker/nginx/network 检查之后添加：

```bash
# 检查 Node.js
if ! command -v node >/dev/null 2>&1; then
  log_error "Node.js 未安装"
  log_error "安装方式: curl -fsSL https://deb.nodesource.com/setup_21.x | sudo -E bash - && sudo apt install -y nodejs"
  return 1
fi
log_info "Node.js: $(node --version)"

# 检查 pnpm
if ! command -v pnpm >/dev/null 2>&1; then
  log_error "pnpm 未安装"
  log_error "安装方式: sudo npm install -g pnpm"
  return 1
fi
log_info "pnpm: $(pnpm --version)"

# 检查 noda-apps 目录
local apps_dir="${1:-$WORKSPACE/noda-apps}"
if [ ! -d "$apps_dir" ]; then
  log_error "noda-apps 目录不存在: $apps_dir"
  log_error "请检查 Jenkinsfile Pre-flight stage 的 checkout 配置"
  return 1
fi

# 检查 package.json 和 lint/test 脚本
if [ ! -f "$apps_dir/package.json" ]; then
  log_error "noda-apps/package.json 不存在: $apps_dir/package.json"
  return 1
fi

if ! grep -q '"lint"' "$apps_dir/package.json"; then
  log_error "noda-apps/package.json 缺少 lint 脚本"
  log_error "请在 package.json 的 scripts 中添加: \"lint\": \"eslint .\""
  return 1
fi

if ! grep -q '"test"' "$apps_dir/package.json"; then
  log_error "noda-apps/package.json 缺少 test 脚本"
  log_error "请在 package.json 的 scripts 中添加: \"test\": \"vitest run\""
  return 1
fi

log_success "所有前置检查通过"
```

注意：pipeline_preflight 需要接受一个可选参数 APPS_DIR（默认 $WORKSPACE/noda-apps），函数签名改为 `pipeline_preflight([APPS_DIR])`。

**2. 优化 pipeline_test() — 添加版本输出和阶段标识**

```bash
pipeline_test() {
  local apps_dir="$1"

  log_info "=========================================="
  log_info "Test 阶段开始"
  log_info "=========================================="

  # 输出环境信息
  log_info "Node.js: $(node --version)"
  log_info "pnpm: $(pnpm --version)"

  # 安装依赖
  log_info "[install] 安装依赖..."
  cd "$apps_dir"
  pnpm install --frozen-lockfile
  log_success "[install] 依赖安装完成"

  # Lint 检查（TEST-01）
  log_info "[lint] 执行代码检查..."
  if pnpm lint; then
    log_success "[lint] 代码检查通过"
  else
    log_error "[lint] 代码检查失败 — Pipeline 中止，不进入部署阶段"
    return 1
  fi

  # 单元测试（TEST-02）
  log_info "[test] 执行单元测试..."
  if pnpm test; then
    log_success "[test] 单元测试通过"
  else
    log_error "[test] 单元测试失败 — Pipeline 中止，不进入部署阶段"
    return 1
  fi

  log_success "=========================================="
  log_success "Test 阶段完成"
  log_success "=========================================="
}
```

关键变更：
- 每步操作有 `[阶段名]` 前缀标识，Jenkins console output 中清晰区分
- lint 失败时打印 "Pipeline 中止，不进入部署阶段" 明确提示
- test 失败时同样打印明确提示
- 版本信息在 Test 阶段开始时输出，便于排查环境问题
