    创建独立的验证脚本，执行以下检查项。脚本应使用 `set -euo pipefail`，输出带状态符号的检查结果。

    ```bash
    #!/bin/bash
    # Phase 6 验证脚本 - 修复变量冲突
    set -euo pipefail

    SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
    LIB_DIR="$SCRIPT_DIR/lib"
    FAILED=0

    echo "=========================================="
    echo "Phase 6 验证：修复变量冲突"
    echo "=========================================="
    echo ""

    # 检查 1: constants.sh 存在且定义了所有退出码
    echo -n "检查 1: constants.sh 存在并定义退出码... "
    if [[ -f "$LIB_DIR/constants.sh" ]] && grep -q '^readonly EXIT_SUCCESS=0' "$LIB_DIR/constants.sh"; then
      echo "✅ 通过"
    else
      echo "❌ 失败"
      FAILED=$((FAILED + 1))
    fi

    # 检查 2: 无重复 EXIT_* 定义（constants.sh 除外）
    echo -n "检查 2: 无重复 readonly EXIT_* 定义... "
    DUPES=$(grep -rn '^readonly EXIT_' "$LIB_DIR"/*.sh | grep -v constants.sh || true)
    if [[ -z "$DUPES" ]]; then
      echo "✅ 通过"
    else
      echo "❌ 失败"
      echo "  发现重复定义: $DUPES"
      FAILED=$((FAILED + 1))
    fi

    # 检查 3: 所有使用 EXIT_* 的库文件有条件加载 constants.sh
    echo -n "检查 3: 所有 EXIT_* 使用者有防御性加载... "
    MISSING_GUARD=0
    for file in db.sh verify.sh cloud.sh health.sh test-verify.sh alert.sh metrics.sh; do
      if grep -q 'return \$EXIT_' "$LIB_DIR/$file" 2>/dev/null; then
        if ! grep -q 'EXIT_SUCCESS+x' "$LIB_DIR/$file"; then
          echo "❌ 失败"
          echo "  $file 使用 EXIT_* 但无条件加载 constants.sh"
          MISSING_GUARD=$((MISSING_GUARD + 1))
        fi
      fi
    done
    if [[ $MISSING_GUARD -eq 0 ]]; then
      echo "✅ 通过"
    else
      FAILED=$((FAILED + 1))
    fi

    # 检查 4: 所有库文件使用 _*_LIB_DIR 前缀（无裸 LIB_DIR）
    echo -n "检查 4: 无裸 LIB_DIR 变量名... "
    BARE_LIB_DIR=$(grep -rn '^\(LIB_DIR=\|readonly LIB_DIR\)' "$LIB_DIR"/*.sh || true)
    if [[ -z "$BARE_LIB_DIR" ]]; then
      echo "✅ 通过"
    else
      echo "❌ 失败"
      echo "  发现裸 LIB_DIR: $BARE_LIB_DIR"
      FAILED=$((FAILED + 1))
    fi

    # 检查 5: 无 .bak 残留文件
    echo -n "检查 5: 无 .bak 残留文件... "
    BAK_FILES=$(find "$LIB_DIR" -name '*.bak' -o -name '*.bak2' 2>/dev/null || true)
    if [[ -z "$BAK_FILES" ]]; then
      echo "✅ 通过"
    else
      echo "❌ 失败"
      echo "  发现残留文件: $BAK_FILES"
      FAILED=$((FAILED + 1))
    fi

    # 检查 6: 所有库文件语法检查通过
    echo -n "检查 6: 所有库文件语法检查... "
    SYNTAX_FAILED=0
    for file in "$LIB_DIR"/*.sh; do
      if ! bash -n "$file" 2>/dev/null; then
        echo "❌ 失败"
        echo "  语法错误: $file"
        SYNTAX_FAILED=$((SYNTAX_FAILED + 1))
      fi
    done
    if [[ $SYNTAX_FAILED -eq 0 ]]; then
      echo "✅ 通过"
    else
      FAILED=$((FAILED + 1))
    fi

    # 检查 7: 主脚本语法检查通过
    echo -n "检查 7: 主脚本语法检查... "
    if bash -n "$SCRIPT_DIR/backup-postgres.sh" 2>/dev/null; then
      echo "✅ 通过"
    else
      echo "❌ 失败"
      FAILED=$((FAILED + 1))
    fi

    # 检查 8: test-verify-weekly.sh 语法检查通过
    echo -n "检查 8: test-verify-weekly.sh 语法检查... "
    if bash -n "$SCRIPT_DIR/test-verify-weekly.sh" 2>/dev/null; then
      echo "✅ 通过"
    else
      echo "❌ 失败"
      FAILED=$((FAILED + 1))
    fi

    # 汇总
    echo ""
    echo "=========================================="
    if [[ $FAILED -eq 0 ]]; then
      echo "✅ 所有检查通过 ($FAILED failed)"
      echo "=========================================="
      exit 0
    else
      echo "❌ $FAILED 项检查失败"
      echo "=========================================="
      exit 1
    fi
    ```
