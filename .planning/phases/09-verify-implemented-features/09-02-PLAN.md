```bash
cat > /tmp/e2e_test.sh << 'EOF'
#!/bin/bash
set -e

echo "=== 集成测试开始 ==="

# 1. 备份
echo "步骤 1: 执行备份"
bash scripts/backup/backup-postgres.sh
backup_exit=$?
echo "备份退出码: $backup_exit"
if [[ $backup_exit -ne 0 ]]; then
  echo "❌ 备份失败"
  exit 1
fi
echo "✅ 备份成功"

# 2. 恢复
echo "步骤 2: 执行恢复"
latest_backup=$(bash scripts/backup/restore-postgres.sh --list-backups | grep keycloak | tail -1 | awk '{print $NF}')
bash scripts/backup/restore-postgres.sh --restore "$latest_backup" --database test_e2e_integration
restore_exit=$?
echo "恢复退出码: $restore_exit"
if [[ $restore_exit -ne 0 ]]; then
  echo "❌ 恢复失败"
  exit 1
fi
echo "✅ 恢复成功"

# 3. 验证
echo "步骤 3: 执行验证测试"
bash scripts/backup/test-verify-weekly.sh --databases keycloak
verify_exit=$?
echo "验证退出码: $verify_exit"
if [[ $verify_exit -ne 0 ]]; then
  echo "❌ 验证失败"
  exit 1
fi
echo "✅ 验证成功"

# 4. 清理
echo "步骤 4: 清理测试数据库"
docker exec noda-infra-postgres-1 psql -U postgres -c "DROP DATABASE test_e2e_integration;" || true
echo "✅ 清理完成"

echo "=== 集成测试完成 ==="
EOF

chmod +x /tmp/e2e_test.sh
bash /tmp/e2e_test.sh
```

**预期结果**:
- 所有 4 个步骤退出码都为 0
- 完整流程无错误
- 测试数据库清理成功

## 🚨 错误处理

### 如果备份失败
**症状**: backup-postgres.sh 退出码非 0

**诊断**:
1. 检查日志：`tail -100 scripts/backup/logs/backup.log`
2. 检查 PostgreSQL：`docker exec noda-infra-postgres-1 pg_isready -U postgres`
3. 检查磁盘空间：`df -h`
4. 检查环境变量：`cat scripts/backup/.env.backup`

**解决**:
- PostgreSQL 连接失败：重启容器
- 磁盘空间不足：清理旧备份
- 环境变量错误：检查 .env.backup

### 如果恢复失败
**症状**: restore-postgres.sh 退出码非 0

**诊断**:
1. 检查 B2 连接：`docker exec opdev rclone ls b2remote:noda-backup`
2. 检查备份文件：`bash scripts/backup/restore-postgres.sh --list-backups`
3. 检查测试数据库：`docker exec noda-infra-postgres-1 psql -U postgres -c "SELECT datname FROM pg_database WHERE datname LIKE 'test_%';"`

**解决**:
- B2 连接失败：检查 rclone 配置
- 备份文件不存在：先执行备份
- 测试数据库冲突：手动删除旧数据库

### 如果验证测试失败
**症状**: test-verify-weekly.sh 退出码非 0

**诊断**:
1. 检查测试日志：`docker exec opdev tail -100 /var/log/noda-backup/test.log`
2. 检查临时数据库：`docker exec noda-infra-postgres-1 psql -U postgres -c "SELECT datname FROM pg_database WHERE datname LIKE 'test_restore_%';"`
3. 检查 B2 下载：`docker exec opdev rclone ls b2remote:noda-backup/postgres/$(date +%Y/%m/%d)`

**解决**:
- 下载失败：检查 B2 连接
- 恢复失败：检查备份文件完整性
- 验证失败：检查验证逻辑
- 清理失败：手动删除临时数据库
